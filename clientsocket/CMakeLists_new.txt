cmake_minimum_required(VERSION 3.10)
project(clientSocket_new VERSION 1.0 LANGUAGES C CXX) 

include(CTest)
enable_testing()


# Explicitly list all source files
set(SOURCES
    main.cpp
    clientSocket_new.cpp
    ../utils/logger.cpp
    ../utils/util.cpp
    ../utils/File.cpp
)

# List headers separately (optional, for IDE visibility)
set(HEADERS
    clientSocket_new.h
    ../utils/logger.h
    ../utils/util.h
    ../utils/File.h
)

if (UNIX)
    set(CMAKE_PREFIX_PATH "../../../vcpkg/installed/x64-windows/share/fmt")
    find_package(fmt CONFIG REQUIRED)
endif()

add_executable(clientSocket_new ${SOURCES} ${HEADERS})
set_property(TARGET clientSocket_new PROPERTY CMAKE_CXX_STANDARD 20)

if(WIN32)
    # Link Winsock library on Windows
    message(STATUS "Compiling for Windows")
    target_link_libraries(clientSocket_new ws2_32)
else()
    # Link pthread library on Unix-like systems
    message(STATUS "Compiling for Linux")
    target_link_libraries(clientSocket_new pthread fmt::fmt) 
endif()
